# Pre-built Chaos Scenarios for SRE Training
# Located in entropy-engine/scenarios/

monday_deployment_failure:
  name: "Monday Deployment Failure"
  description: "Simulates common deployment issues that occur during Monday evening releases"
  duration: "30 minutes"
  learning_objectives:
    - "Correlation between deployments and error rates"
    - "Cascading failure patterns"
    - "SLO impact assessment"
  
  timeline:
    - time: "0m"
      action: "increase_error_rate"
      target: "payment-service"
      value: "15%"
      reason: "New payment provider integration has bugs"
      
    - time: "2m"
      action: "cascade_latency"
      target: "ecommerce-api"
      value: "increase_by_500ms"
      reason: "Payment timeouts cause checkout delays"
      
    - time: "5m"
      action: "queue_backlog"
      target: "job-processor"
      value: "throughput_50%"
      reason: "Failed payments generate retry jobs"
      
    - time: "15m"
      action: "auto_recover"
      target: "all"
      reason: "Rollback deployment completed"

  expected_alerts:
    - "PaymentServiceErrorRate > 10%"
    - "CheckoutLatencyHigh > 2s" 
    - "JobProcessorQueueDepth > 500"
    - "SLOBreach: CheckoutSLO < 90%"

  monitoring_focus:
    - "Multi-window SLO burn rate"
    - "Service dependency mapping"
    - "Error budget consumption"

  expected_behavior:
    - "Circuit breaker prevents further damage"

    - "Increased latency but maintained availability"
    - "Queue depth increases but stays manageable"
